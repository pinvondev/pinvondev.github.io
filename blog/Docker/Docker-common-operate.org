* 查看在使用的容器

#+BEGIN_SRC Shell
docker ps -a
#+END_SRC

* 关掉所有在使用的容器

#+BEGIN_SRC Shell
docker ps -qa | xargs docker rm
# 或
docker rm $(docker ps -qa)

# 如果无法删除成功
docker rm --force $(docker ps -qa)
#+END_SRC

* 删除所有镜像文件

#+BEGIN_SRC Shell
docker rmi $(docker images -q)
#+END_SRC
