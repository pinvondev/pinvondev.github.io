查看在使用的容器

docker ps -a

关掉所有在使用的容器

docker ps -qa | xargs docker rm
# 或
docker rm $(docker ps -qa)

# 如果无法删除成功
docker rm --force $(docker ps -qa)

删除所有镜像文件

docker rmi $(docker images -q)
