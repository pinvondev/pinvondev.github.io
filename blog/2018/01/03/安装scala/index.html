安装Scala

>>>>>>> origin/master 
sudo tar -zxf scala-2.12.4.tgz -C /usr/local
cd /usr/local
sudo mv ./scala-2.12.4 ./scala
sudo chown -R pinvon:pinvon ./scala

emacs ~/.bashrc

export PATH=$PATH:/usr/local/scala/bin

source ~/.bashrc

# 测试
scala
